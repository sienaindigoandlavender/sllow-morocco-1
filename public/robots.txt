# Default — search engines, humans, retrieval/citation crawlers.
# NOTE: do not add a separate "User-agent: Googlebot / Allow: /" group.
# A named group REPLACES the * group for that bot, which silently
# cancels every Disallow below for Googlebot. That is exactly what
# let ~10k opengraph-image renders and /api/ endpoints back into the
# crawl queue. One group, applied to everyone, is the correct shape.
User-agent: *
Allow: /
Disallow: /admin
Disallow: /api/
# OG share-card images: noindex assets, not pages. Blocking them stops
# Googlebot from spending crawl budget on ~10k dynamic image renders
# (which intermittently 5xx under rapid crawling and clutter Search Console).
Disallow: /opengraph-image
Disallow: /twitter-image
Disallow: /*/opengraph-image
Disallow: /*/twitter-image

# Retrieval / citation crawlers visit on demand and link users back to
# the source. They are covered by the * group above — welcome by default.
